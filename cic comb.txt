module cic_comb
#(
    parameter WIDTH = 32
)
(
    input wire clk,
    input wire rst_n,
    input wire valid,

    input wire signed [WIDTH-1:0] din,

    output reg signed [WIDTH-1:0] dout
);

reg signed [WIDTH-1:0] delay;

always @(posedge clk or negedge rst_n)
begin
    if(!rst_n)
    begin
        delay <= 0;
        dout  <= 0;
    end
    else if(valid)
    begin
        dout  <= din - delay;
        delay <= din;
    end
end

endmodule
