module cic_integrator
#(
    parameter IN_WIDTH  = 32,
    parameter OUT_WIDTH = 32
)
(
    input  wire clk,
    input  wire rst_n,

    input  wire signed [IN_WIDTH-1:0] din,

    output reg signed [OUT_WIDTH-1:0] dout
);

always @(posedge clk or negedge rst_n)
begin
    if(!rst_n)
        dout <= 0;
    else
        dout <= dout + din;
end

endmodule
